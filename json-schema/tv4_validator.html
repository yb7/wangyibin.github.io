﻿## 通过tv4进行JSON Schema验证 tv4是一个高效的JSON Schema验证库。

var tv4 = require('tv4');
var _ = require('lodash');


/**
 * 定义数据更新的Schema
 * additionalProperties: false 代表不支持未定义的属性
 **/
var schemaForUpdate = {
  id: 'cityupdate#',
  additionalProperties: false,
  properties: {
    code: {type: 'string'},
    name: {type: 'string'},
    pinyin: {type: 'string'},
    latitude: {type: 'number'},
    longitude: {type: 'number'},
    woeid: {type: 'number'},
    enable: {type: 'boolean'},
    hot: {type: 'boolean'},
    ads: {
      type: 'object',
      properties: {
        top: {
          type: 'array',
          items: { $ref: '#/definitions/adItem' }
        }
      }
    }
  },
  definitions: {
    adItem: {
      type: 'object',
      required: ['imageKey', 'description'],
      additionalProperties: false,
      properties: {
        imageKey: {type: 'string'},
        description: {type: 'string'}
      }
    }
  }
};

/**
 * 定义数据创建的Schema
 * 在更新的Schema基础上增加required属性
 * 即在数据创建时 'code', 'name', 'pinyin' 三个字段为必填
 **/
var schemaForCreate = _.defaults({
  id: 'citycreate#',
  required: ['code', 'name', 'pinyin']
}, schemaForUpdate);

/**
 * 将两个Schema加入到tv4中
 **/
tv4.addSchema(schemaForCreate);
tv4.addSchema(schemaForUpdate);

var Schema = module.exports = {};

/**
 * 导出tv4的json验证
 * 这里使用validateMultiple意思是执行所有的数据验证，收集尽可能多的错误信息
 **/
Schema.validateForCreate = function(instance) {
  return tv4.validateMultiple(instance, 'citycreate#');
};
Schema.validateForUpdate = function(instance) {
  return tv4.validateMultiple(instance, 'cityupdate#');
};

Test Case书写如下：

var assert = require('assert');

var CitySchema = require('../common/models/city.schema');

describe('city model', function(){
  describe('#schema', function(){
    it('should validate city create', function() {
      var validCityJson = {
        code: 'xx',
        name: 'xx',
        pinyin: 'xx',
        latitude: 1.0,
        longitude: 1.0,
        woeid: 10,
        enable: true,
        hot: true
      };
      assert.equal(CitySchema.validateForCreate(validCityJson).valid, true, '正确的格式');
      assert.equal(CitySchema.validateForCreate({}).valid, false, '必填项必须填写');
      assert.equal(CitySchema.validateForCreate({x:3}).valid, false, '不允许未定义的属性');
    });
    it('should validate city update', function() {
      var validUpdateJson = {
        pinyin: 'xx',
        latitude: 1.0,
        longitude: 1.0,
        woeid: 10
      };
      assert.equal(CitySchema.validateForUpdate(validUpdateJson).valid, true, '正确的格式');
      assert.equal(CitySchema.validateForUpdate({ads:{top:[{imageKey:'ee'}]}}).valid, false, '广告条目必须完整');
      assert.equal(CitySchema.validateForUpdate({ads:{top:[{imageKey:'ee', description:'xx'}]}}).valid, true);
      assert.equal(CitySchema.validateForUpdate({x:3}).valid, false, '不允许未定义的属性');
    });
  });
});
