loopback v2.8.0 当前仅支持mongodb的2d index。执行流程如下：

当查询条件中包含如下查询信息时

{
  "geolocation": {
    "near": [50, 30],
    "maxDistance": 1
  }
}

查询loopback-datasource-juggler/dao.js的find方法中的执行流程如下。

// DataAccessObject.find
DataAccessObject.find = function find(query, options, cb) {
  // ...
  /** 当query.where中有key的值中包含near这个字段时，
   *  geo.nearFilter会构建地理位置查询语句。
   *  但该查询语句是针对mongodb 的2d index的比较过时
   **/
  var near = query && geo.nearFilter(query.where);
  var supportsGeo = !!this.getDataSource().connector.buildNearFilter;


  if (near) {
    if (supportsGeo) {
      // 如果connector中存在buildNearFilter方法，则通过该方法进行优化处理
      // 但在 loopback-connector-mongodb版本1.8.0中并没有实现改方法
      this.getDataSource().connector.buildNearFilter(query, near);
    } else if (query.where) {
      // 当connector中未实现buildNearFilter是，会通过内存方式进行数据过滤
      // do in memory query
      // 也就是说不会进行实际的数据查询
      return;
    }

    // ...
    self.getDataSource().connector.all(self.modelName, query, allCb);
}

所以总结如下，near查询在loopback-connector-mongodb 1.8.0 中不会被真正的翻译 为mongodb的$near语句，所以我们需要进行优化。

优化思路如下:

  1. 使用新的查询关键字xnear替换原有的near以绕过geo.nearFilter
  2. 当geo.nearFilter无法构建near查询对象时，则绕过了内存数据查询
  3. 通过Model.observe('access')在执行实际的查询前构建适用于mongodb的地理位置查询语句
  4. 执行mongodb的$near查询

查询条件调整如下

{
  "geolocation": {
    "xnear": [50, 30],
    "maxDistance": 1
  }
}

构建QueryRefine工具类queryrefine.js

var debug = require('debug')('oversea:query-refine');

var QueryRefine = module.exports = {};

QueryRefine.refine = function(query) {
  var where = query.where;
  debug('Before Refine: ', where);
  if (where && typeof where === 'object') {
    Object.keys(where).forEach(function (key) {
      var ex = where[key];

      if (ex && ex.xnear) {
        where[key] = {
          'near': {
            '$geometry' :{
              'type' : 'Point' ,
              'coordinates': ex.xnear
            }
          }
        };
        if (ex.maxDistance) {
          where[key].near.$maxDistance = ex.maxDistance;
        }
        debug('Refine Geo near: ', where);
      }
    });
  }
};

在Model对象上拦截查询，并优化geo参数

Merchant.observe('access', function refineQuery(ctx, next) {
  queryRefine.refine(ctx.query);
  next();
});
