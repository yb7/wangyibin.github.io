MongoDB.prototype.all = function all(model, filter, callback) {
  // ...
  var query = {};
  if (filter.where) {
    query = self.buildWhere(model, filter.where);
  }
  // ...
}

MongoDB.prototype.buildWhere = function (model, where) {
  var self = this;
  var query = {};
  // 如果没有where这个字段就直接返回空
  if (where === null || (typeof where !== 'object')) {
    return query;
  }
  var idName = self.idName(model);
  Object.keys(where).forEach(function (k) {
    var cond = where[k];
    if (k === idName) {
      k = '_id';
      cond = ObjectID(cond);
    }
    if (k === 'and' || k === 'or' || k === 'nor') {
      if (Array.isArray(cond)) {
        cond = cond.map(function (c) {
          return self.buildWhere(model, c);
        });
      }
      query['$' + k ] = cond;
      delete query[k];
      return;
    }
    var spec = false;
    var options = null;

    //如果查询的key所对应的值是个对象，那么需要提取将该对象的第一个key作为spec
    if (cond && cond.constructor.name === 'Object') {
      options = cond.options;
      spec = Object.keys(cond)[0];
      cond = cond[spec];
    }
    if (spec) { // 对于传多值的类型按照key的类型组装查询语句
      if (spec === 'between') {
        query[k] = { $gte: cond[0], $lte: cond[1]};
      } else if (spec === 'inq') {
        query[k] = { $in: cond.map(function (x) {
          if ('string' !== typeof x) return x;
          return ObjectID(x);
        })};
      } else if (spec === 'like') {
        query[k] = {$regex: new RegExp(cond, options)};
      } else if (spec === 'nlike') {
        query[k] = {$not: new RegExp(cond, options)};
      } else if (spec === 'neq') {
        query[k] = {$ne: cond};
      }
      else {
        query[k] = {};
        query[k]['$' + spec] = cond;
      }
    } else { // 对于简单的单值查询，直接将键值对拷贝到查询语句中
      if (cond === null) {
        // http://docs.mongodb.org/manual/reference/operator/query/type/
        // Null: 10
        query[k] = {$type: 10};
      } else {
        query[k] = cond;
      }
    }
  });
  return query;
}

Loopback仅提供了非常有限的通用查询方法，有时候我们必须借助原生的查询语句来完成数据的访问。

因此实现该目的我们有以下两种写法

方法一

Merchant.findMerchantsNeedUpdateIndex = function(cb) {
  Merchant.find({where: {$where: 'this.searchableTextUpdateTime > this.searchIndexUpdateTime'}}, cb);
};

方法二

Merchant.findMerchantsNeedUpdateIndex = function(cb) {
    //获取Mongodb的数据库链接
    var mongodbConnection = Merchant.dataSource.connector.db;
    if (!mongodbConnection) {
      // 有可能数据库链接还未建立，这里需要手动去建立链接
      // 注意这个是异步操作
      Merchant.dataSource.connect(function(err, db) {
        mongodbConnection = db;
      });
    }
    // 通过 mongodbConnection.collection('Merchant') 获取Merchant的集合
    mongodbConnection.collection('Merchant').find({$where: 'this.searchableTextUpdateTime > this.searchIndexUpdateTime'}).
