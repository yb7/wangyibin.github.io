可以在Model.js文件中调用validates__Of方法来定义模型验证方法

// 定义商户的名称长度不能少于4个字符
Merchant.validatesLengthOf('name', {min: 4});

// 商户的名称必须唯一
Merchant.validatesUniquenessOf('name');

// 验证邮箱的格式
var emailRegex = /^(([^<>()[\]\\.,;:\s@\']+(\.[^<>()[\]\\.,;:\s@\']+)*)|(\'.+\'))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/; // jshint ignore:line
Merchant.validatesFormatOf('email', {with: emailRegex, message: 'Must provide a valid email'});

// 验证手机号码的格式
var telephoneRegex = /^\(\d+\)(\d+-)?\d+$/;
Merchant.validatesFormatOf('telephone', {with: telephoneRegex, message: 'Must provide a valid telephone'});
